### 10.合并数组

```js
let a1 = ['A1', 'A2', 'B1', 'B2', 'C1', 'C2', 'D1', 'D2']
let a2 = ['A', 'B', 'C', 'D']
//合并为
// ['A1','A2','A','B1','B2','B','C1','C2','C','D1','D2','D']

let res1 = []
while (a1.length || a2.length) {
  if (a1.length == 0) {
    res1 = [...res1, ...a2]
    break
  }
  if (a2.length == 0) {
    res1 = [...res1, ...a1]
    break
  }
  if (a2[0] == a1[0][0]) {
    res1.push(a1.shift())
  } else {
    res1.push(a2.shift())
  }
}
console.log('--', res1)
```
